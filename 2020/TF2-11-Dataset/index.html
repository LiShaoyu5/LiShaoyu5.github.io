# 2.2.0

tf.data是TensorFlow中用于构建输pipeline的一组Dataset的API。

Dataset：TensorFlow中用于表示数据的类，可以看作是数据的有序列表。它有三个常用子类：

  • FixedLengthRecordDataset: 用于从二进制文件中读取定长记录的Dataset。
  • TFRecordDataset: 从一个或多个TFRecord文件中读取并组成记录的Dataset。
  • TextLineDataset: 从一个或多个文本文件中读取并组成记录的Dataset。

1. Dataset的基本使用

建立Dataset，有三种读取数据的方法：

  • from_tensors: 读取Tensor并合并，创建只有单个元素的Dataset。
  • from_tensor_slices: 将读入的Tensor按第一维分割，创建有多个元素的Dataset。
  • from_generator: 读取一个generator（可以调用iter()方法的对象），创建一个使用此generator生成数据的Dataset。
import numpy as np

data1 = np.zeros(shape=(3,5,2),dtype=np.float32)
data2 = tf.constant([[1, 2], [3, 4]])

从Tensor读取数据，可以看出from_tensors和from_generator的区别：

# 合并元素，建立具有单个元素的Dataset

dataset1 = tf.data.Dataset.from_tensors(data1)
print('dataset1:')
for line in dataset1:
    print(line.shape)  # 只有1个元素，形状为3*5*2

dataset2 = tf.data.Dataset.from_tensors(data2)
print('\ndataset2:')
for line in dataset2:
    print(line)  # 只有1个元素，形状为2*2

# 以输入tensor的第一维分割，建立具有多个元素的Dataset

dataset1 = tf.data.Dataset.from_tensor_slices(data1)
print('dataset1:')
for line in dataset1:
    print(line.shape)  # 3*5*2的数据被分割为3个5*2的数据

dataset2 = tf.data.Dataset.from_tensor_slices(data2)
print('\ndataset2:')
for line in dataset2:
    print(line)  # 2*2的数据被分割为2个1*2的数据

因此，读取数据集一般使用from_tensors方法：

# 从numpy读取数据

mnist = np.load('./data/mnist.npz')
x_train, y_train = mnist['x_train'],mnist['y_train']
x_train = np.expand_dims(x_train, axis=-1)
print(x_train.shape, y_train.shape)


# 输入数据为元组，在迭代时也可以成对地读取
mnist_dataset = tf.data.Dataset.from_tensor_slices((x_train, y_train))
import matplotlib.pyplot as plt
for image, label in mnist_dataset:
    plt.title(label.numpy())
    plt.imshow(image.numpy()[:, :,0])
    plt.show()
    break

可以看出，x_train和y_train均被分割为60000个Tensor，每个表示一张28*28的单通道图像对应一个label。

对一个建立好的Dataset，常用到以下处理函数：

  • Dataset.map(f): 对其中所有元素应用函数f，得到新的数据集。
  • Dataset.shuffle(buffer_size): 打乱数据集。
  • Dataset.batch(batch_size): 将数据集分成批次。

2. TFRecordDataset的使用

TFRecordDataset常用于处理体积过大而无法一次性读入内存的数据集，TensorFlow能高效、快速地处理.tfrecords格式的文件。

# 定义Feature结构，告诉解码器每个Feature的类型是什么
feature_description = {
    'image': tf.io.FixedLenFeature([], tf.string),
    'label': tf.io.FixedLenFeature([], tf.int64),
}

# 将TFRecord文件中的每一个序列化的tf.train.Example解码
def _parse_example(example_string):
    feature_dict = tf.io.parse_single_example(example_string, feature_description)
    feature_dict['image'] = tf.io.decode_jpeg(feature_dict['image'])    # 解码JPEG图片
    feature_dict['image'] = tf.image.resize(feature_dict['image'], [256, 256]) / 255.0
    return feature_dict['image'], feature_dict['label']

train_dataset = tf.data.TFRecordDataset("./data/tfdata/sub_train.tfrecords")    # 读取 TFRecord 文件

train_dataset = train_dataset.map(_parse_example)

for image,label in train_dataset:
    print(image)
    print(label)
    break

3. TextLineDataset类的使用

TextLineDataset提供了从文本文件读取生成数据集的简单方法，它将文本文件的一行视为一个Tensor。

titanic_lines = tf.data.TextLineDataset(['./data/tfdata/train.csv','./data/tfdata/eval.csv'])

# 将一行string按，分开
def data_func(line):
    line = tf.strings.split(line, sep = ",")
    return line

titanic_data = titanic_lines.skip(1).map(data_func)

for line in titanic_data:
    print(line)
    break

4. Dataset中的常用操作

# flat_map(f)：将f映射到数据集中的每个元素，再将数据集展平
dataset = tf.data.Dataset.from_tensor_slices([[1, 2, 3], [4, 5, 6]])

print('dataset:')
for i in dataset:
    print(i)

print('datasetq:')
dataset2 = dataset.flat_map(lambda x: tf.data.Dataset.from_tensor_slices(x))
for i in dataset2:
    print(i)

# interleave(f, cycle_length, block_length)：每次取出cycle_length个元素，应用f，然后每次放回block_length个直到此次结果全部放回
dataset = tf.data.Dataset.range(1, 6)  # [1, 2, 3, 4, 5]
dataset = dataset.interleave(
    lambda x: tf.data.Dataset.from_tensors(x).repeat(3),
    cycle_length=2, block_length=2)
list(dataset.as_numpy_iterator())

# zip：将两个数据集纵向连接
a = tf.data.Dataset.range(1, 4)
b = tf.data.Dataset.range(4, 7)

print('dataset1:')
dataset1 = tf.data.Dataset.zip((a, b))
for i in dataset1:
    print(i)

print('dataset2:')
dataset2 = tf.data.Dataset.zip((b, a))
for i in dataset2:
    print(i)

# concatenate：将两个数据集横向连接
a = tf.data.Dataset.range(1, 4)
b = tf.data.Dataset.range(4, 7)

print('dataset1:')
dataset1 = tf.data.Dataset.concatenate(a, b)
for i in dataset1:
    print(i)

print('dataset2:')
dataset2 = tf.data.Dataset.concatenate(b, a)
for i in dataset2:
    print(i)

5. Dataset提升性能的方法

  • prefech()：将准备数据与参数迭代同时执行。
  • interleave()：可以并行化读取数据。
  • map()：在函数中指定num_parallel_calls参数，使其中函数的应用并行执行。
  • cache：将数据集缓存，避免某些操作在每个epoch都执行。一般在map()后添加。

## 6.2 构建模型

```Python
class CNNModel(tf.keras.models.Model):
    def __init__(self):
        super(CNNModel, self).__init__()
        self.conv1 = tf.keras.layers.Conv2D(32, 3, activation='relu')
        self.maxpool1 = tf.keras.layers.MaxPooling2D()
        self.conv2 = tf.keras.layers.Conv2D(32, 1, activation='relu')
        self.maxpool2 = tf.keras.layers.MaxPooling2D()
        self.flatten = tf.keras.layers.Flatten()
        self.d1 = tf.keras.layers.Dense(64, activation='relu')
        self.d2 = tf.keras.layers.Dense(2, activation='softmax')

    def call(self, x):
        x = self.conv1(x)
        x = self.maxpool1(x)
        x = self.conv2(x)
        x = self.maxpool2(x)       
        x = self.flatten(x)
        x = self.d1(x)
        x = self.d2(x)
        return x

model = CNNModel()

loss_object = tf.keras.losses.SparseCategoricalCrossentropy()  #label 没有one-hot

optimizer = tf.keras.optimizers.Adam(learning_rate=0.001)

train_loss = tf.keras.metrics.Mean(name='train_loss')
train_accuracy = tf.keras.metrics.SparseCategoricalAccuracy(name='train_accuracy')

test_loss = tf.keras.metrics.Mean(name='test_loss')
test_accuracy = tf.keras.metrics.SparseCategoricalAccuracy(name='test_accuracy')


@tf.function
def train_step(images, labels):
    with tf.GradientTape() as tape:
        predictions = model(images)
        loss = loss_object(labels, predictions)
    gradients = tape.gradient(loss, model.trainable_variables)
    optimizer.apply_gradients(zip(gradients, model.trainable_variables))

    train_loss(loss)
    train_accuracy(labels, predictions)

def test_step(images, labels):
    predictions = model(images)
    t_loss = loss_object(labels, predictions)

    test_loss(t_loss)
    test_accuracy(labels, predictions)

6.3 训练模型

EPOCHS = 5
for epoch in range(EPOCHS):
    # 在下一个epoch开始时，重置评估指标
    train_loss.reset_states()
    train_accuracy.reset_states()
    test_loss.reset_states()
    test_accuracy.reset_states()

    for images, labels in train_dataset:
        train_step(images, labels)

    for images, labels in test_dataset:
        test_step(images, labels)

    template = 'Epoch {}, Loss: {}, Accuracy: {}, Test Loss: {}, Test Accuracy: {}'
    print(template.format(epoch + 1,
                          train_loss.result(),
                          train_accuracy.result() * 100,
                          test_loss.result(),
                          test_accuracy.result() * 100
